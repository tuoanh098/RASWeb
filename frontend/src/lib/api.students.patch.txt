// Thêm vào src/lib/api.js
export const Api = {
  // ...giữ nguyên các mục khác
  students: {
    list: (params) => http('/api/students', { params }),
    get: (id) => http(`/api/students/${id}`),
    create: (payload) => http('/api/students', { method:'POST', body: payload }),
    update: (id, payload) => http(`/api/students/${id}`, { method:'PUT', body: payload }),
    delete: (id) => http(`/api/students/${id}`, { method:'DELETE' })
  }
}
